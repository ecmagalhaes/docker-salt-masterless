FROM ubuntu:16.04
MAINTAINER "Paolo Smiraglia <paolo.smiraglia@gmail.com>"

# install dependencies
RUN apt-get update \
    && apt-get install -y -q \
        curl \
        git \
        vim

# add SaltStack repo
ADD conf/saltstack.list /etc/apt/sources.list.d/saltstack.list

# install SaltStack
RUN curl https://repo.saltstack.com/apt/ubuntu/16.04/amd64/latest/SALTSTACK-GPG-KEY.pub | apt-key add - \
    && apt-get update \
    && apt-get install -y -q \
        salt-minion
